tree ::= <statement_list>

condition ::= <expression>[==|!=|<|<=|>|>=]<expression>|<expression>

directive ::= <directive_bank>|<directive_data>|<directive_define>|<directive_if>|<directive_if_define>|<directive_include>
		| <directive_origin>|<directive_reserve>|<directive_undefine>

directive_bank ::= .bank<expression>

directive_data ::= .[db|dw]<expression_list>

directive_define ::= .def<identifier><expression>

directive_else ::= .else<statement_list>

directive_else_if ::= .elsif<condition><statement_list>

directive_end ::= .end

directive_if ::= .if<condition><statement_list><directive_else_if>*<directive_else>?<directive_end>

directive_if_define ::= .ifdef<identifier><statement_list><directive_else>?<directive_end>

directive_include ::= .[inc|incb]<literal>

directive_origin ::= .org<expression>

directive_reserve ::= .res<expression><expression>

directive_undefine ::= .undef<identifier>

expression ::= <expression_factor>[+|-|*|/|%|&|||^|&&||||<<|>>]<expression>|<expression_factor>

expression_factor ::= [~|!|-]?<macro>?[([<identifier>|<label>|<literal>]([<expression>])?)|<scalar>]

expression_list ::= <expression>,<expression_list>|<expression>

instruction ::= <instruction_ccf>|<instruction_cpl>|<instruction_daa>|<instruction_di>|<instruction_ei>
	|<instruction_halt>|<instruction_nop>|<instruction_ret>|<instruction_reti>|<instruction_rla>
	|<instruction_rlca>|<instruction_rra>|<instruction_rrca>|<instruction_scf>|<instruction_stop>

instruction_ccf ::= ccf

instruction_cpl ::= cpl

instruction_daa ::= daa

instruction_di ::= di

instruction_ei ::= ei

instruction_halt ::= halt

instruction_nop ::= nop

instruction_ret ::= ret

instruction_reti ::= reti

instruction_rla ::= rla

instruction_rlca ::= rlca

instruction_rra ::= rra

instruction_rrca ::= rrca

instruction_scf ::= scf

instruction_stop ::= stop

statement ::= <directive>|<instruction>|<label>|<register>

statement_list ::= <statement>*
